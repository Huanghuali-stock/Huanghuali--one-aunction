// Price Data for Chart.js
const priceHistoryData = {
  labels: ['Ming Dynasty', 'Qing Dynasty', '20th Century', '21st Century'],
  datasets: [{
    label: 'Huanghuali Auction Price ($)',
    data: [5000, 20000, 100000, 2000000], // Sample data for historical prices
    backgroundColor: 'rgba(54, 162, 235, 0.2)',
    borderColor: 'rgba(54, 162, 235, 1)',
    borderWidth: 1
  }]
};

// Configure Chart.js to display price history
const ctx = document.getElementById('priceChart').getContext('2d');
const priceChart = new Chart(ctx, {
  type: 'line',
  data: priceHistoryData,
  options: {
    scales: {
      y: {
        beginAtZero: true
      }
    }
  }
});
